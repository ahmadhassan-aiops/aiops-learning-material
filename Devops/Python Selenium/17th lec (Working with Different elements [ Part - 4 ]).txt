import time
from selenium import webdriver
from webdriver_manager.firefox import GeckoDriverManager
from selenium.webdriver.common.by import By


driver = webdriver.Firefox(executable_path=GeckoDriverManager().install())
driver.get("https://admin-demo.nopcommerce.com/Admin/Order/List")
driver.maximize_window()

my_email = driver.find_element(By.ID,"Email")
time.sleep(2)
my_email.clear()
time.sleep(2)
my_email.send_keys('admin@yourstore.com')
time.sleep(2)
my_pass = driver.find_element(By.ID,"Password")
my_pass.clear()
my_pass.send_keys('admin')


driver.find_element(By.XPATH,'/html/body/div[6]/div/div/div/div/div[2]/div[1]/div/form/div[3]/button').click()

time.sleep(3)

def is_elements_avail(att,abd):
    if len(driver.find_elements(by=att,value=abd)) == 0:
        return False
    else:
        return True
i= 1
while is_elements_avail(By.XPATH,'//*[@id="orders-grid"]/tbody/tr['+str(i)+']/td[1]/input'):
    driver.find_element(By.XPATH,'//*[@id="orders-grid"]/tbody/tr['+str(i)+']/td[1]/input').click()
    time.sleep(1)
    i = i + 1
driver.quit()

import time
from selenium import webdriver
from webdriver_manager.firefox import GeckoDriverManager
from selenium.webdriver.common.by import By


driver = webdriver.Firefox(executable_path=GeckoDriverManager().install())
driver.get("https://admin-demo.nopcommerce.com/Admin/Order/List")
driver.maximize_window()

my_email = driver.find_element(By.ID,"Email")
time.sleep(2)
my_email.clear()
time.sleep(2)
my_email.send_keys('admin@yourstore.com')
time.sleep(2)
my_pass = driver.find_element(By.ID,"Password")
my_pass.clear()
my_pass.send_keys('admin')


driver.find_element(By.XPATH,'/html/body/div[6]/div/div/div/div/div[2]/div[1]/div/form/div[3]/button').click()

time.sleep(3)

my_checkboxes = driver.find_elements(By.CLASS_NAME,"checkboxGroups")
for i in my_checkboxes:
    i.click()
    time.sleep(0.2)
print("Total checkbox value is : ",len(driver.find_elements(By.CLASS_NAME,"checkboxGroups")))
driver.quit()
