Webdriver code.py

import time
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager

# Initialize the Chrome driver with the correct service
service = Service(ChromeDriverManager().install())
driver = webdriver.Chrome(service=service)

# Open the IMDb website
driver.get("https://www.imdb.com/")
driver.maximize_window()

# Keep the browser open for 6 seconds
time.sleep(6)

# Verify the title
web_title = driver.title
if web_title == "IMDb: Ratings, Reviews, and Where to Watch the Best Movies & TV Shows":
    print(f"Yes, we verified and the title name is: {web_title}")
else:
    print("Sorry, title not found from alnafi database")

# Close the browser
driver.quit()
