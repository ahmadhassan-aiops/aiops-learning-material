import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager

# Initialize Chrome WebDriver with Service
service = Service(ChromeDriverManager().install())
driver = webdriver.Chrome(service=service)

# Navigate to the Wikipedia homepage
driver.get("https://www.wikipedia.org/")
driver.maximize_window()

# Find all <option> elements in the dropdown
options = driver.find_elements(By.TAG_NAME, "option")

# Print the number of options
print(f"Number of options: {len(options)}")

# Optional: Print the text of each option
for option in options:
    print(option.text)

# Wait for a few seconds and quit the browser
time.sleep(5)
driver.quit()

import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.select import Select
from webdriver_manager.firefox import GeckoDriverManager


driver = webdriver.Firefox(executable_path=GeckoDriverManager().install())

driver.get("https://dev.alnafi.com/")
driver.maximize_window()

a_tag = driver.find_element(By.XPATH,'//*[@id="__nuxt"]/div/footer/div[1]/div[2]/div[2]')
anchors_tag = a_tag.find_elements(By.TAG_NAME,"a")

mylinks=[]
for links in anchors_tag:
    print("Text is : ",links.text, "Link is :", links.get_attribute("href"))
    mylinks.append(links.text)
print(len(anchors_tag))
print("4th Link-----------------------------")
print(mylinks[3])
print("------------------------------------")

driver.quit()
