in this lecture we will use data types like list to store the name of packages to install 

# Valid for Centos
sudo vim packages.yml

---
- name: this yml file is dedicated to installation of packages
  hosts: centos
  become: yes
  vars:
    packages: ["wget","zip","curl"]
  tasks:
  - name: installation of packages using yum
    yum: 
      name: {{ packages }}
      state: present

vim list_pack_instl.yml

---
- name: this yml file is dedicated to installation of packages
  hosts: ubuntu
  become: yes
  vars:
    packages: ["wget", "zip", "curl"]
  tasks:
  - name: installation of packages using apt
    apt:
      name: "{{ packages }}"
      state: present
      update_cache: yes
