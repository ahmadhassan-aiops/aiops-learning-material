import os
import datetime

find_file=input("Enter the search:")

for roots,dirs,files in os.walk("/root/lect25/"):
    for i in files:
        if i==find_file:
            mypath=(os.path.join(roots,i))
            mycustomdate=datetime.datetime.fromtimestamp(os.path.getctime(mypath))
            mydate=(mycustomdate.strftime("%B %d %Y"))
            mytime=(mycustomdate.strftime("%I:%M:%S %p"))
            print(f"We found the file {mypath} : \n {mypath} date is : {mydate} \n {mypath} time is : {mytime}")
