#cat /etc/os-release

import os
myversion=os.popen("cat /etc/os-release").read()
mylist=myversion.split()

#print(mylist)
myfind=(mylist[8:10])
convert_string=" ".join(myfind)
#print(convert_string)
myversion=(convert_string.strip("\""))
print(f"My OS version is : {myversion} ")
