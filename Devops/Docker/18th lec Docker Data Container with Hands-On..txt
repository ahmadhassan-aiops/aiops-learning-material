Lecture 14 --> Same name directory , directory create but you cant access your existing data from another container


docker create -v /dbdata --name dbstore postgres:latest /bin/true (IT WILL WORK)
docker run -d --volumes-from dbstore --name db1 -e POSTGRES_PASSWORD=mysecretpassword postgres:latest


dbdata --> db1
docker exec -it db1 /bin/sh
dbdata --> db2

docker run -d --volumes-from dbstore --name db2 -e POSTGRES_PASSWORD=mysecretpassword postgres:latest
docker exec -it db2/bin/sh
